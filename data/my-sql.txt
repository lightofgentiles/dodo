CREATE TABLE `banner` (
  `banner_num` int(11) NOT NULL AUTO_INCREMENT,
  `banner_desc` varchar(1000) DEFAULT NULL,
  `banner_url` varchar(200) DEFAULT NULL,
  `product_id` int(11) DEFAULT NULL,
  PRIMARY KEY (`banner_num`)
) 

CREATE TABLE `board` (
  `board_num` int(11) NOT NULL AUTO_INCREMENT,
  `board_id` varchar(50) DEFAULT NULL,
  `board_subject` varchar(100) DEFAULT NULL,
  `board_content` varchar(2000) DEFAULT NULL,
  `board_re_ref` int(11) DEFAULT NULL,
  `board_re_lev` int(11) DEFAULT NULL,
  `board_re_seq` int(11) DEFAULT NULL,
  `board_readcount` int(11) DEFAULT NULL,
  `board_date` date DEFAULT NULL,
  `board_parent` int(11) DEFAULT NULL,
  PRIMARY KEY (`board_num`)
) 

CREATE TABLE `cart` (
  `cart_id` int(11) NOT NULL AUTO_INCREMENT,
  `user_id` varchar(20) DEFAULT NULL,
  `product_id` int(11) NOT NULL,
  `amount` int(11) DEFAULT '0',
  PRIMARY KEY (`cart_id`),
  KEY `cart_product_fk` (`product_id`),
  KEY `cart_userid_fk` (`user_id`),
  CONSTRAINT `cart_product_fk` FOREIGN KEY (`product_id`) REFERENCES `product` (`product_id`),
  CONSTRAINT `cart_userid_fk` FOREIGN KEY (`user_id`) REFERENCES `member` (`user_id`)
)

CREATE TABLE `delivery` (
  `order_num` int(11) NOT NULL,
  `order_name` varchar(100) DEFAULT NULL,
  `order_address` varchar(200) DEFAULT NULL,
  PRIMARY KEY (`order_num`),
  CONSTRAINT `delivery_ibfk_1` FOREIGN KEY (`order_num`) REFERENCES `orderlist` (`order_num`)
)

CREATE TABLE `member` (
  `user_id` varchar(20) NOT NULL,
  `user_pass` varchar(20) DEFAULT NULL,
  `user_name` varchar(20) DEFAULT NULL,
  `user_email` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`user_id`)
)

CREATE TABLE `orderlist` (
  `order_num` int(11) NOT NULL DEFAULT '0',
  `user_id` varchar(20) DEFAULT NULL,
  `order_date` date DEFAULT NULL,
  `order_money` int(11) DEFAULT NULL,
  `address` varchar(200) DEFAULT NULL,
  `tele` varchar(100) DEFAULT NULL,
  `order_state` int(11) DEFAULT NULL,
  PRIMARY KEY (`order_num`),
  KEY `user_id` (`user_id`),
  CONSTRAINT `orderlist_ibfk_1` FOREIGN KEY (`user_id`) REFERENCES `member` (`user_id`)
) 

CREATE TABLE `orders` (
  `order_num` int(11) DEFAULT NULL,
  `user_id` varchar(20) DEFAULT NULL,
  `order_amount` int(11) DEFAULT NULL,
  `order_name` varchar(50) DEFAULT NULL,
  `order_money` int(11) DEFAULT NULL,
  KEY `user_id` (`user_id`),
  CONSTRAINT `orders_ibfk_1` FOREIGN KEY (`user_id`) REFERENCES `member` (`user_id`)
)

CREATE TABLE `product` (
  `product_id` int(11) NOT NULL AUTO_INCREMENT,
  `product_num` int(11) DEFAULT NULL,
  `product_name` varchar(50) DEFAULT NULL,
  `product_price` int(11) DEFAULT '0',
  `product_desc` varchar(500) DEFAULT NULL,
  `product_url` varchar(500) DEFAULT NULL,
  PRIMARY KEY (`product_id`)
)

CREATE TABLE `qaboard` (
  `board_num` int(11) NOT NULL AUTO_INCREMENT,
  `board_id` varchar(50) DEFAULT NULL,
  `board_subject` varchar(100) DEFAULT NULL,
  `board_content` varchar(2000) DEFAULT NULL,
  `board_re_ref` int(11) DEFAULT NULL,
  `board_re_lev` int(11) DEFAULT NULL,
  `board_re_seq` int(11) DEFAULT NULL,
  `board_readcount` int(11) DEFAULT NULL,
  `board_date` date DEFAULT NULL,
  `board_parent` int(11) DEFAULT NULL,
  PRIMARY KEY (`board_num`)
)

CREATE TABLE `sequence` (
  `seq_num` int(11) DEFAULT NULL,
  `seq_name` varchar(30) DEFAULT NULL,
  `seq` int(11) DEFAULT '0'
)
